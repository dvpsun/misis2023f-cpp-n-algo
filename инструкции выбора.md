if-инструкция
• if (x)  //… 
  означает 
  if (0 != x) //… 
  
• if (!x) //… 
  означает 
  if (0 == x) //…

if-инструкция (пример)
if (0 == isReady) { 
  // основная работа 
} else { 
  // обработка дополнительных случаев 
}

switch-инструкция 
• значение сопоставляется с набором констант 
• можно заменить набором if-инструкций

switch-инструкция (пример) 
switch (val) {
case 1: 
  fA(); 
  break; 
case 2: 
  fB(); 
  break;
default: 
  onError(); 
  break; 
}
